    <?php

/**
 * 权重随机轮训
 */
class RandomWeight
{
    /**
     * @var array 原数据
     */
    private $items;

    /**
     * @var string 原数据中的权重key
     */
    private $weightKey = 'weight';

    /**
     * 设置原数据
     * @param array $data
     * @return void
     * @author Junduo <caijunduo@gmail.com>
     */
    public function setItem(array $data)
    {
        $this->items = $data;
    }

    /**
     * 设置原数据中的权重key
     * @param string $weightKey
     * @return void
     * @author Junduo <caijunduo@gmail.com>
     */
    public function setWeightKey(string $weightKey)
    {
        $this->weightKey = $weightKey;
    }

    /**
     * 权重轮训，输出对应的item
     * @return mixed|void
     * @author Junduo <caijunduo@gmail.com>
     */
    public function roundRobin()
    {
        if (!$this->items) {
            return;
        }
        if (count($this->items) == 1) {
            return current($this->items);
        }

        $totalWeight = $this->getTotalWeight();

        if ($totalWeight <= 0) {
            return;
        }

        $randNum = rand(1, $totalWeight);

        foreach ($this->items as $key => $val) {
            if ($randNum <= $val[$this->weightKey]) {
                return $val;
            }
            $randNum -= $val[$this->weightKey];
        }
    }

    /**
     * 获取总权重
     * @return float|int
     * @author Junduo <caijunduo@gmail.com>
     */
    private function getTotalWeight()
    {
        $weights = array_column($this->items, $this->weightKey);
        return array_sum($weights);
    }
}
